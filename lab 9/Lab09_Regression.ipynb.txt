    "## Task 3.3: Polynomial Curve Fitting using x1\n",
    "\n",
    "We'll try polynomial degrees from 2 to 11 and pick the one with lowest validation RMSE.\n"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "id": "2c2b10ba",
   "metadata": {},
   "outputs": [],
   "source": [
    "degrees = list(range(2, 12))  # 2, 3, ..., 11\n",
    "train_rmses_poly = []\n",
    "val_rmses_poly   = []\n",
    "models_poly      = []\n",
    "\n",
    "for p in degrees:\n",
    "    poly = PolynomialFeatures(degree=p)\n",
    "    \n",
    "    X_train_poly = poly.fit_transform(X_train_x1)\n",
    "    X_val_poly   = poly.transform(X_val_x1)\n",
    "    \n",
    "    model = LinearRegression()\n",
    "    model.fit(X_train_poly, y_train)\n",
    "    \n",
    "    tr_rmse  = rmse(y_train, model.predict(X_train_poly))\n",
    "    val_rmse = rmse(y_val,   model.predict(X_val_poly))\n",
    "    \n",
    "    train_rmses_poly.append(tr_rmse)\n",
    "    val_rmses_poly.append(val_rmse)\n",
    "    models_poly.append((model, poly))\n",
    "    \n",
    "    print(f\"Degree {p:2d} | Train RMSE: {tr_rmse:.4f} | Val RMSE: {val_rmse:.4f}\")\n"
   ]
