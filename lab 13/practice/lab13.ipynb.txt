    "def purity_score(y_true, y_pred):\n",
    "    \"\"\"\n",
    "    Compute clustering purity score.\n",
    "    For each cluster, find the majority true label; purity = sum of majority counts / total points.\n",
    "    Noise points (label == -1) are included in a separate 'noise' cluster.\n",
    "    \"\"\"\n",
    "    from collections import Counter\n",
    "    total = len(y_true)\n",
    "    correct = 0\n",
    "    unique_clusters = set(y_pred)\n",
    "    for cluster in unique_clusters:\n",
    "        mask = (y_pred == cluster)\n",
    "        cluster_labels = y_true[mask]\n",
    "        if len(cluster_labels) == 0:\n",
    "            continue\n",
    "        most_common_count = Counter(cluster_labels).most_common(1)[0][1]\n",
    "        correct += most_common_count\n",
    "    return correct / total"
   ]
